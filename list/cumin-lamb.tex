\recipe{
    Cumin Lamb
}{
    Josh Brown
}{
    \Ingredients{
        \item Cooking oil
        \item 2 Tbsp Sichuan peppercorns
        \item 2 dried chili peppers
        \item 8 cloves of garlic, minced
        \item 3 green onions
        \item 2 Tbsp cumin
        \item 1 lb Lamb, thinly sliced
    }
}{
    \Directions{
        \item Heat the oil over medium heat in a pan.
            Add the peppercorns and chili peppers; cook until fragrant.
        \item Add the garlic and the bottom thirds of the green onions, finely chopped.
            Cook about 2 minutes.
            Add a light dusting of cumin and cook for another 2-3 minutes.
        \item Add the lamb and the rest of the cumin.
            Stir once to coat evenly, and then stirring occasionally until the lamb is cooked.
        \item Add the upper parts of the green onions, and serve.
    }
}
