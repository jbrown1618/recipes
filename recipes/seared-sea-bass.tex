\Recipe{
    Seared Sea Bass over Canellini Beans
}{
    \href{https://www.cookinglight.com/recipes/seared-sea-bass-lemon-olive-white-beans}{CookingLight}
}{
    \Ingredients{
        \item 4 sea bass fillets
        \item \threefourths \teaspoon kosher salt 
        \item \half \teaspoon black pepper 
        \item 3 \tablespoons extra-virgin olive oil 
        \item 5 garlic cloves, sliced 
        \item 2 \teaspoons fresh thyme leaves 
        \item 8 cherry tomatoes, quartered
        \item 2/3 \cup unsalted chicken stock 
        \item 1 15 \ounce can unsalted cannellini beans, rinsed and drained
        \item 5 \ounces fresh baby spinach 
        \item 2 \tablespoons fresh lemon juice 
        \item 1 \ounce pitted Castelvetrano olives
    }
}{
    \Directions{
        \item Heat a large nonstick skillet over medium-high.
            Sprinkle fish with \fourth \teaspoon salt and \fourth \teaspoon pepper.
            Add 1 \tablespoon oil to skillet; swirl to coat.
        \item Add fish; cook to desired degree of doneness, about 3 minutes on each side for medium.
            Remove fish from pan; discard drippings.
            Do not wipe skillet clean.
            Lightly tent fish with foil to keep warm.
        \item Heat remaining 2 \tablespoons oil in skillet over medium-high.
        \item Add garlic; cook, stirring often, until light golden brown, about 1 minute.
            Stir in thyme and tomatoes; cook, stirring often, until heated through, about 1 minute.
        \item Add stock and beans; bring to a simmer.
            Add spinach in batches, and cook, tossing gently, until spinach is wilted after each addition.
        \item Stir in lemon juice, olives, remaining \half \teaspoon salt, and remaining \fourth \teaspoon pepper.
        \item Divide bean mixture among 4 shallow bowls; nestle 1 fillet into each bowl.
    }
}
