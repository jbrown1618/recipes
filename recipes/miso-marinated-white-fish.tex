\Recipe{
    Miso marinated white fish
}{
    \href{https://hikarimiso.com/recipes/entrees/miso-marinated-fish/}{Hikari Miso}
}{
    \Ingredients{
        \item 2 salmon or black cod fillets
        \item 3 \tablespoons. white miso
        \item 2 \tablespoons. mirin \(sweet sake\)
        \item 2 \tablespoons. sake
        \item 1 \tablespoon. sugar
        \item 1 \tablespoon. soy sauce
    }
}{
    \Directions{
        \item Mix the ingredients for the marinade
        \item Cover each fish fillet in the marinade and seal tightly in a container.
            Refrigerate for 1--2 days.
        \item Remove the fillets and dab or wipe off the excess marinade.
        \item Cook the fish over high heat.
    }
}
