\Recipe{placeholder}{
    Granola
}{
    Liz Shively
}{
    \Ingredients{
        \item 4 \cups rolled oats
        \item 1 \cup wheat germ
        \item \half \cup flax seed meal
        \item \half \cup brown sugar
        \item \half \cup raw sunflower seeds
        \item \half \cup chopped walnuts
        \item \half \cup chopped almonds
        \item 1 \teaspoon ground cinnamon
        \item 1 \teaspoon salt
        \item \third \cup canola oil
        \item \fourth \cup honey
        \item 1 \teaspoon vanilla extract
        \item \half \cup water

    }
}{
    \Directions{
        \item Preheat oven to 300 degrees F.
        \item In a large bowl, mix together first nine ingredients.
            In a smaller bowl, whisk together oil, honey, vanilla, and water.
            Pour wet ingredients over dry and mix until evenly blended.
        \item Spread in a greased 9x13 or 11x7 inch baking pan.
        \item Bake for 1 hour in preheated oven, stirring every 20 minutes, until granola is toasted.
            Let cool completely before storing in an airtight container.

    }
}
