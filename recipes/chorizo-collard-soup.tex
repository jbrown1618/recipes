\Recipe{placeholder}{
    Chorizo and Collard Greens Soup
}{
    \href{https://www.foodandwine.com/recipes/collard-greens-with-black-eyed-peas}{Bobby Flay}
}{
    \Ingredients{
        \item 3 \half cups chicken stock
        \item 1 onion, chopped
        \item Salt
        \item Pepper
        \item 2 \half pounds collard greens
        \item 2 \pounds chorizo
        \item 1 15 \ounce can black-eyed peas, drained
        \item 2 tablespoons rice vinegar
    }
}{
    \Directions{
        \item In a large saucepan, combine 2 \half \cups of the stock with the 
            onion and a generous pinch each of salt and pepper and bring to a boil.
        \item While the stock is coming to a boil, prepare the collards:
            strip the leaves from the stems, roughly chop the leaves, and finely chop the stems.
        \item Add the collards in large handfuls, letting each batch wilt slightly before adding more.
        \item Cover and cook over moderate heat, stirring occasionally, until the collards are just
            tender, about 25 minutes.
        \item Uncover and add the chorizo in chunks.
            Cook, stirring occasionally, until the chorizo is cooked and the broth is slightly reduced, about 10 minutes.
        \item Meanwhile, in a small saucepan, combine the black-eyed peas with the remaining 1 \cup of
            stock and bring to a boil.
        \item Simmer the black-eyed peas over moderate heat for 8 minutes; season with salt and pepper.
            Using a slotted spoon, add the black-eyed peas to the collards.
        \item Add the vinegar to the collard greens and season with salt and pepper.
    }
}
