\Recipe{placeholder}{
    Roasted Lamb Rack (孜然烤羊排)
}{
    \href{https://www.xiachufang.com/recipe/100294156/}{xiachufang}
}{
    \Ingredients{
        \item 1 lamb rack
        \item 3 chunks of ginger
        \item 1 cinnamon stick
        \item 1 star anise
        \item 1 \tablespoon Sichuan peppercorns
        \item cooking oil
        \item 2 \teaspoons salt
        \item 1 \tablespoon cumin
    }
}{
    \Directions{
        \item Preheat the oven to 400 degrees F.
        \item Place the lamb rack in a pressure cooker.
            Add ginger, cinnamon, star anise, peppercorns, and enough water to submerge the lamb.
            Start the pressure cooker over high heat.
            Once it is up-to-pressure, reduce the heat to medium for 20 minutes.
        \item Remove the lamb. Drain, and let cool.
        \item Rub the oil, salt, and cumin over the lamb.
        \item Cover with aluminum foil, and bake for 20 minutes.
        \item To color and crisp the outer skin, remove the aluminum foil, and broil for 5 minutes.
    }
}
