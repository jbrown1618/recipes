\Recipe{
    Corn Pudding
}{
    \textit{The Joy of Cooking}
}{
    \Ingredients{
        \item 2 \cups fresh or canned corn kernels, drained
        \item \threefourths \cup milk or half-and-half
        \item 2 large eggs, well beaten
        \item 2 \tablespoons butter, melted and cooled
        \item 1 \tablespoon all-purpose flour
        \item 1 \teaspoon salt
    }
}{
    \Directions{
        \item Preheat the oven to 350 degrees F.
        \item Butter a 1 \half quart baking dish
        \item Mix all ingredients together in a large bowl.
        \item Pour into the prepared baking dish.
        \item Place in a larger pan of water and bake for 30--45 minutes, until the center is set.
    }
}
