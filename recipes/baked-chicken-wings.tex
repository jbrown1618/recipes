\Recipe{placeholder}{
    Baked Chicken Wings
}{
    \href{https://www.dinneratthezoo.com/baked-chicken-wings/}{Dinner at the Zoo}
}{
    \Ingredients{
        \item 1 pound chicken wings
        \item 1 \tablespoon brown sugar
        \item 1 \tablespoons paprika
        \item 1 \half \teaspoon smoked paprika
        \item 1 \half \teaspoon kosher salt
        \item 1 \half \teaspoon garlic powder
        \item 1 \half \teaspoon onion powder
        \item 1 \teaspoon black pepper
        \item \half \teaspoon mustard powder
        \item \quarter \teaspoon cayenne pepper
    }
}{
    \Directions{
        \item Preheat oven to 400 degrees F. Line a baking sheet with aluminum foil.
        \item Mix the spices together in a large plastic bag.
        \item Shake the wings in the bag with the spice mixture until they are evenly coated.
        \item Place the wings skin-side-down on the baking sheet, and bake for 20 minutes.
        \item Flip the wings skin-side-up and bake for another 25 minutes.
        \item Broil for an additional minute as needed, to desired crispiness.
    }
}
