\Recipe{placeholder}{
    Pancakes
}{
    \textit{The Joy of Cooking}
}{
    \Ingredients{
        \item 1 \half \cups all purpose flour
        \item 2 \tablespoons sugar
        \item 1 \teaspoon salt
        \item 3 \tablespoons butter, melted
        \item 2 eggs
    }
    \NamedIngredients{Buttermilk}{
        \item 1 \half \tablespoons baking powder
        \item \half \teaspoon baking soda
        \item 1 \half \cups buttermilk
    }
    \NamedIngredients{Regular}{
        \item 1 \half \tablespoons baking powder
        \item 1 \quarter \cups milk
    }
}{
    \Directions{
        \item Mix together the dry ingredients and wet ingredients in separate bowls.
        \item Grease a skillet lightly with butter or oil and do so as needed between batches.
            The heat is correct when a few drops of cold water on the skillet bounce and sputter, not boiling or evaporating.
        \item Mix the liquid quickly into the dry ingredients.
        \item Use \fourth \cup batter for each pancake. When bubbles appear on the surface on the pancake and the edges have browned, turn the cake and cook only until the second side is done.
    }
}
