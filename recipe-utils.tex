\newcommand{\NamedIngredients}[2]{
    \subsection*{#1}
    \begin{itemize} 
        #2
    \end{itemize}
}

\newcommand{\Ingredients}[1]{
    \NamedIngredients{Ingredients}{#1}
}

\newcommand{\NamedDirections}[2]{
    \subsection*{#1}
    \begin{enumerate} 
        #2
    \end{enumerate}
}

\newcommand{\Directions}[1]{
    \NamedDirections{Directions}{#1}
}

\newcommand{\Image}[1]{
    \includegraphics[width=\textwidth]{img/#1.png}
}

\newcommand{\Caption}[1]{

}

\newcommand{\Title}[2]{
    \section*{ \hspace{-12pt} #1 {\hfill \normalsize \normalfont from #2}}
    \addcontentsline{toc}{section}{ \hspace{-6pt} #1 }
}

\newcommand{\PageSplit}[2]{
    \begin{minipage}[t]{0.35\textwidth}
        #1
    \end{minipage}
    \hspace{0.05\textwidth}
    \begin{minipage}[t]{0.6\textwidth}
        #2
    \end{minipage}
}
